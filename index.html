public class Main {
    public enum Sex { MALE, FEMALE };

// Эта программа определяет, есть ли у человека право запросить кредит. Вот правила, по которым она работает:
// человек должен быть не моложе 21 года
// мужчина должен быть моложене старше 65 лет на момент прекращения срока кредита
// женщина должна быть моложене старше 60 лет на момент прекращения срока кредита


    public static void main(String[] args) {
        System.out.print ("Loan possible for female with age 50 and for period 2 years: " +
                isLoanPossible(Sex.FEMALE, 50, 2));
    }

    public static boolean isLoanPossible(Sex sex, int age, int loanPeriod) {
        boolean retval = false;
        if (sex == Sex.FEMALE) {
            if ((age + loanPeriod) < 65) {
                retval = true;
            }
        } else if (sex == Sex.MALE) {
            if ((age + loanPeriod) < 60) {
                retval = true;
            }
        } else if (age < 21) {
            retval = false;
        }
        return retval;

}
